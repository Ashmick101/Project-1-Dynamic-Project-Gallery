// Open the Console to See the Output let a = 10; console.log("Initial value of a:", a); console.log("\n--------------------------\n"); // Arithmetic operations let sum = a + 5; console.log("Addition (a + 5):", sum); let difference = a - 3; console.log("Subtraction (a - 3):", difference); let product = a * 2; console.log("Multiplication (a * 2):", product); let quotient = a / 2; console.log("Division (a / 2):", quotient); let remainder = a % 3; console.log("Modulus (a % 3):", remainder); console.log("\n==========================\n"); // Working with another variable (b) let b = 10; console.log("Initial value of b:", b); console.log("\n--------------------------\n"); // Assignment operators b += 5; // Equivalent to b = b + 5 console.log("Addition Assignment (b += 5):", b); console.log("\n"); b -= 2; // Equivalent to b = b - 2 console.log("Subtraction Assignment (b -= 2):", b); console.log("\n"); b *= 3; // Equivalent to b = b * 3 console.log("Multiplication Assignment (b *= 3):", b); console.log("\n"); b /= 2; // Equivalent to b = b / 2 console.log("Division Assignment (b /= 2):", b); console.log("\n"); b %= 4; // Equivalent to b = b % 4 console.log("Modulus Assignment (b %= 4):", b); console.log("\n==========================\n");
